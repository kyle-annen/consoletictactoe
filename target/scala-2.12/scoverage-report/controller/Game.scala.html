128       x => if (x == boardMove.toString) userToken else x
129     )
130 
131     val gameOver: Boolean = Board.gameOver(updatedBoard)
132 
133     if(gameOver) {

135       val endBoard: List[List[String]] = View.formatBoard(updatedBoard)

139 
140       if (isWin) {
141         View.renderDialog(output, leftPadding, playerNumAnnounce, dialogLang("win"))
142       } else {
143         View.renderDialog(output, leftPadding, dialogLang("tie"))
144       }
145 
146       View.renderWhitespace(output, 5)
147       //this is the return value of the game
148       Map(currentPlayer -> isWin)
149     } else {
150       val nextPlayer: Int = if(currentPlayer == 1) 2 else 1
151       val newLoopCount = loopCount + 1
152 
153       go(
154         updatedBoard, players, dialogLang, false, nextPlayer, output,
155         leftPadding, whiteSpace, getInput, newLoopCount)
156     }
157   }

178         board, players, gameLang, false, currentPlayer,
