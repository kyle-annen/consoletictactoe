122   @tailrec def go(
123     board: List[String],
124     players: Map[Int, (String, String, String)],
125     dialogLang: Map[String, String],
126     gameOver: Boolean,
127     currentPlayer: Int,
128     output: String => Any,

144     val inputPrompt: String = dialogLang("inputPrompt")

148     val difficulty: String = players(currentPlayer)._3

151     val boardMove = if(playerType == "human") {

153       val humanPlay: String = IO.getValidMove(

155         inputPrompt,
156         invalidPlay,
157         output,
158         getInput,
159         leftPadding,

164       val compPlay: Int = AI.getComputerMove(
165         board,
166         userToken,
167         oppToken,
168         userToken,
169         ttTable,
170         difficulty) + 1
171       compPlay
172     }
173 
174     val updatedBoard: List[String] = board.map(
175       x => if (x == boardMove.toString) userToken else x
176     )
177 
178     val gameOver: Boolean = Board.gameOver(updatedBoard)
179 
180     if(gameOver) {
181       View.renderWhitespace(output, whiteSpace)
182       val endBoard: List[List[String]] = View.formatBoard(updatedBoard)
183       View.renderBoard(output, endBoard, leftPadding)
184       View.renderDialog(output, leftPadding, dialogLang("gameOver"))
185       val isWin: Boolean = Board.checkWin(updatedBoard)
186 
187       if (isWin) {
188         View.renderDialog(output, leftPadding, playerNumAnnounce, dialogLang("win"))
189       } else {
190         View.renderDialog(output, leftPadding, dialogLang("tie"))
191       }
192 
193       View.renderWhitespace(output, 5)
194       //this is the return value of the game
195       Map(currentPlayer -> isWin)
196     } else {
197       val nextPlayer: Int = if(currentPlayer == 1) 2 else 1
198       val newLoopCount = loopCount + 1
199 
200       go(
201         updatedBoard, players, dialogLang, false, nextPlayer, output,
202         leftPadding, whiteSpace, getInput, newLoopCount, ttTable)
203     }
204   }
