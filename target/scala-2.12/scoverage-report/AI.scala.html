17   def getBoardTranspositions(
18     board: List[String],
19     score: Int,
20     p1Token: String,
21     p2Token: String): List[(String, Int)] = {
22 
23     val generalizedBoard = board.map { x =>
24       x match {
25         case `p1Token` => p1Token
26         case `p2Token` => p2Token
27         case _ => "-"
28       }
29     }
30     val rowLists = Board.returnRows(generalizedBoard)
31     val b1 = rowLists.flatten.foldLeft("")(_ + _)
32     val b2 = rowLists.transpose.flatten.foldLeft("")(_ + _)
33     val b3 = b1.reverse
34     val b4 = b2.reverse
35 
36     val transpositions = List(
37       (b1 -> score),
38       (b2 -> score),
39       (b3 -> score),
40       (b4 -> score)
41     )
42     transpositions
43   }

55   def checkTransposition(
56     boardState: List[String],
57     ttTable: TranspositionTable,
58     p1Token: String,
59     p2Token: String,
60     minOrMax: String): (Boolean, Int) = {
61     val generalizedBoard = boardState.map {x =>
62       if(x == p1Token) {
63         p1Token
64       } else if(x == p2Token) {
65         p2Token
66       } else {
67         "-"
68       }
69     }
70     val path = if(minOrMax == "min") ttTable.min else ttTable.max
71     val boardKey = generalizedBoard.foldLeft("")(_ + _)
72     val keyPresent = path.contains(boardKey)
73     if(keyPresent) {
74       val score = path(boardKey)
75       (true, score)
76     } else {
77       (false, 0)
78     }
79   }
